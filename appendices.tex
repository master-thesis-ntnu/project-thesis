\section{Elasticsearch Query}
\begin{lstlisting}[language=json, caption={Initial query to retrive query expansion data}, label={ap:initial-query}]
{
    query : {
        terms : {
          tags : [...]
        }
    },
    size: 50,
    aggregations : {
        terms: {
            terms : {
                field : "tags.keyword",
                size: 50
            }
        },
        term_count: {
            value_count: {
                field: "tags.keyword"
            }
        }
    }
}
\end{lstlisting}
